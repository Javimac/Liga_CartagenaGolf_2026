                if (window.lucide && window.lucide.icons) {

                    const icon = window.lucide.icons[name];

                    if (icon && Array.isArray(icon[1])) {

                        try {

                            const paths = icon[1].map(i => {

                                const tagName = i[0];

                                const attrs = i[1] || {};

                                const attrString = Object.entries(attrs)

                                    .map(([k, v]) => `${k}="${v}"`)

                                    .join(' ');

                                return `<${tagName} ${attrString} />`;

                            }).join('');

                           

                            const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">${paths}</svg>`;

                            setIconHtml(svg);

                        } catch (e) {

                            console.warn("Icon error:", name);

                        }

                    }

                }

            }, [name]);

            return <span className={`inline-flex items-center justify-center ${className}`} dangerouslySetInnerHTML={{ __html: iconHtml }} />;

        };

            return `${fixCase(name)} ${fixCase(surnames)}`;

                    } else { data = players.find(x => x.id === tid); ref = publicDataPath.collection('players').doc(tid); }
